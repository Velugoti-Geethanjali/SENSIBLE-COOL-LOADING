    <label class="input-label">Area of the building (m²):</label>
    <input type="number" class="input-field" id="area" step="0.01">
    <label class="input-label">Number of occupants:</label>
    <input type="number" class="input-field" id="occupants">
    <label class="input-label">Type of building:</label>
    <select class="input-field" id="buildingType">
        <option value="residential">Residential</option>
        <option value="commercial">Commercial</option>
    </select>
    <label class="input-label">Outdoor temperature (°C):</label>
    <input type="number" class="input-field" id="outdoorTemp" step="0.01">
    <label class="input-label">Indoor desired temperature (°C):</label>
    <input type="number" class="input-field" id="indoorTemp" step="0.01">
    <button onclick="calculateCoolingLoad()">Calculate Cooling Load</button>
    <p id="result"></p>

    <script>
        function calculateCoolingLoad() {
            var area = parseFloat(document.getElementById("area").value);
            var occupants = parseInt(document.getElementById("occupants").value);
            var buildingType = document.getElementById("buildingType").value;
            var outdoorTemp = parseFloat(document.getElementById("outdoorTemp").value);
            var indoorTemp = parseFloat(document.getElementById("indoorTemp").value);
            var uValue = 30;

            var coolingLoad = (buildingType === "residential") ? 100 * occupants : (buildingType === "commercial") ? 150 * occupants : 0;
            var qConduction = uValue * area * (outdoorTemp - indoorTemp);
            var sensibleCoolingLoad = qConduction + coolingLoad;

            document.getElementById("result").innerHTML = "Sensible Cooling Load: " + sensibleCoolingLoad + " W";
        }
    </script>
